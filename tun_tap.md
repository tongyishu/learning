#### tun&tap

tun&tap是Linux内核的虚拟网卡设备。

tun是三层设备，处理IP数据包，没有MAC地址; tap是二层设备，处理以太帧，有MAC地址。

tun&tap通过读写/dev/net/tun设备，用于在用户空间和内核空间之间传递数据（ip数据包/以太帧）。



#### 物理网卡的报文收发流程?

```bash
                    +-----------------+
                    |     应用程序     |          用户空间
                    +-----------------+
                          |     ^
                          | CPU |
                          v     |
                +-------------------------+
                |        内核协议栈         |     内核空间
                +-------------------------+
                          |     ^
                          | DMA |
                          v     |
                    +-----------------+
                    |     物理网卡     |          物理空间
                    +-----------------+
```



#### 虚拟网卡的报文收发流程?

```
      +----------------------------+
      |   应用程序（openVPN、qemu）  |          用户空间
      +----------------------------+
                 |     ^
                 | CPU |
                 v     |
      +----------------------------+
      |     虚拟网卡（tun、tap）     |    ----> 此处即为tun/tap的应用，通过读写/dev/net/tun设备，
      +----------------------------+          用于用户空间和内核空间的通信。
                 |     ^
                 | CPU |
                 v     |
+---------------------------------------+
|              内核协议栈                 |     内核空间
+---------------------------------------+
                 |     ^
                 | DMA |
                 v     |
           +-----------------+
           |     物理网卡     |                 物理空间
           +-----------------+
```

#### tun&tap实验

```c
#include <fcntl.h>
#include <linux/if_tun.h>
#include <net/if.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/ioctl.h>
#include <sys/stat.h>
#include <sys/types.h>

int tun_alloc(int flags)
{
    struct ifreq ifr;
    int fd, err;
    char* clonedev = "/dev/net/tun";

    if ((fd = open(clonedev, O_RDWR)) < 0) {
        return fd;
    }

    memset(&ifr, 0, sizeof(ifr));
    ifr.ifr_flags = flags;

    if ((err = ioctl(fd, TUNSETIFF, (void*)&ifr)) < 0) {
        close(fd);
        return err;
    }

    printf("Open tun/tap device: %s for reading...\n", ifr.ifr_name);
    return fd;
}

int main()
{
    int tun_fd, nread;
    char buffer[1500];

    /* Flags: IFF_TUN   - TUN device (no Ethernet headers)
     *        IFF_TAP   - TAP device
     *        IFF_NO_PI - Do not provide packet information
     */
    tun_fd = tun_alloc(IFF_TAP | IFF_NO_PI);
    if (tun_fd < 0) {
        perror("Allocating interface");
        exit(1);
    }

    while (1) {
        nread = read(tun_fd, buffer, sizeof(buffer));
        if (nread < 0) {
            perror("Reading from interface");
            close(tun_fd);
            exit(1);
        }

        printf("Read %d bytes from tun/tap device\n", nread);
    }
    return 0;
}
```
